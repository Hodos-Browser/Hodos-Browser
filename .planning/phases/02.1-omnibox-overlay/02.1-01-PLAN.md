```cpp
void CreateOmniboxOverlay(HINSTANCE hInstance) {
    LOG_INFO_APP("🔍 Creating omnibox overlay");

    // Check if overlay already exists
    extern HWND g_omnibox_overlay_hwnd;
    if (g_omnibox_overlay_hwnd && IsWindow(g_omnibox_overlay_hwnd)) {
        LOG_WARNING_APP("🔍 Omnibox overlay already exists, showing it");
        ShowWindow(g_omnibox_overlay_hwnd, SW_SHOW);
        SetForegroundWindow(g_omnibox_overlay_hwnd);
        return;
    }

    // Get main window position
    extern HWND g_hwnd;
    RECT mainRect;
    GetWindowRect(g_hwnd, &mainRect);
    int width = mainRect.right - mainRect.left;

    // Omnibox overlay: positioned at top, 300px height for dropdown
    int overlayHeight = 300;

    // Register window class if not already registered
    WNDCLASSEX wc = {0};
    wc.cbSize = sizeof(WNDCLASSEX);
    if (!GetClassInfoEx(hInstance, L"CEFOmniboxOverlayWindow", &wc)) {
        wc.style = CS_HREDRAW | CS_VREDRAW;
        wc.lpfnWndProc = DefWindowProc;
        wc.hInstance = hInstance;
        wc.hCursor = LoadCursor(nullptr, IDC_ARROW);
        wc.lpszClassName = L"CEFOmniboxOverlayWindow";
        RegisterClassEx(&wc);
    }

    // Create overlay window (TOPMOST and LAYERED like wallet)
    HWND omnibox_hwnd = CreateWindowEx(
        WS_EX_LAYERED | WS_EX_TOOLWINDOW | WS_EX_TOPMOST,
        L"CEFOmniboxOverlayWindow",
        L"Omnibox Overlay",
        WS_POPUP | WS_VISIBLE,
        mainRect.left, mainRect.top, width, overlayHeight,
        g_hwnd, nullptr, hInstance, nullptr);

    if (!omnibox_hwnd) {
        LOG_ERROR_APP("❌ Failed to create omnibox overlay HWND");
        return;
    }

    g_omnibox_overlay_hwnd = omnibox_hwnd;

    // Set transparency (75% opaque background)
    SetLayeredWindowAttributes(omnibox_hwnd, 0, 192, LWA_ALPHA);

    // Enable mouse input
    EnableWindow(omnibox_hwnd, TRUE);

    // Create CEF browser in overlay
    CefWindowInfo window_info;
    window_info.SetAsChild(omnibox_hwnd, CefRect(0, 0, width, overlayHeight));

    CefRefPtr<SimpleHandler> handler = new SimpleHandler("omnibox_overlay");
    CefBrowserSettings settings;
    std::string url = "http://127.0.0.1:5137/omnibox-overlay";

    CefBrowserHost::CreateBrowser(
        window_info,
        handler,
        url,
        settings,
        nullptr,
        CefRequestContext::GetGlobalContext()
    );

    LOG_INFO_APP("✅ Omnibox overlay created");
}
```

```typescript
import React, { useEffect, useRef } from 'react';
import { Box } from '@mui/material';
import Omnibox from '../components/Omnibox';

const OmniboxOverlayRoot: React.FC = () => {
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    console.log('🔍 Omnibox overlay mounted');

    // Focus will be handled by Omnibox component's onFocus
  }, []);

  const handleNavigate = (url: string) => {
    console.log('🔍 Navigating to:', url);

    // Send navigate message via IPC
    if (window.cefMessage) {
      window.cefMessage.send('omnibox_navigate', [url]);
    }
  };

  const handleEscape = () => {
    console.log('🔍 Closing omnibox overlay');

    // Send close message via IPC
    if (window.cefMessage) {
      window.cefMessage.send('omnibox_close', []);
    }
  };

  // Listen for Escape key at container level
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === 'Escape') {
        handleEscape();
      }
    };

    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, []);

  return (
    <Box
      ref={containerRef}
      sx={{
        width: '100%',
        height: '100%',
        display: 'flex',
        flexDirection: 'column',
        alignItems: 'center',
        pt: 2,
        px: 2,
        bgcolor: 'transparent',
      }}
    >
      {/* Unified pill container wrapping the entire omnibox */}
      <Box
        sx={{
          width: '100%',
          maxWidth: '800px',
          borderRadius: 20,
          bgcolor: '#ffffff',
          boxShadow: '0 4px 16px rgba(0, 0, 0, 0.2)',
          overflow: 'visible',
        }}
      >
        <Omnibox
          onNavigate={handleNavigate}
          initialValue=""
        />
      </Box>
    </Box>
  );
};

export default OmniboxOverlayRoot;
```
